const projects = projectsRaw.map(p => {
    const loc = p.fr.location.toLowerCase();
    const isNational = loc.includes('tunisie') || loc.includes('tunisia') || loc.includes('bizerte') || loc.includes('kasserine') || loc.includes('siliana') || loc.includes('medjerda') || loc.includes('ariana') || loc.includes('sidi bouzid') || loc.includes('kairouan') || loc === 'tunisie' || loc === 'nord tunisie';
    return { ...p, scope: isNational ? 'national' : 'international' };
});

let currentLang = 'fr';
let currentCategory = 'water';
let currentScope = null;

function renderProjects() {
    const filtered = projects.filter(p => {
        if (p.category !== currentCategory) return false;
        if (currentScope && p.scope !== currentScope) return false;
        return true;
    });
    const grid = document.getElementById('projectsGrid');
    grid.innerHTML = filtered.map((p, idx) => {
        const data = p[currentLang];
        return `<div class="project-card" data-idx="${idx}"><div class="project-image" style="background-image:url('${p.img}'); background-size:cover; background-position:center;"></div><div class="project-content"><span class="project-category">${p.category === "water" ? (currentLang === 'fr' ? "Hydraulique" : "Hydraulics") : p.category === "environment" ? (currentLang === 'fr' ? "Environnement" : "Environment") : (currentLang === 'fr' ? "Infrastructures" : "Infrastructure")}</span><h3 class="project-title">${data.title}</h3><p class="project-client">${data.client}</p><p>${data.shortDesc.substring(0, 100)}${data.shortDesc.length > 100 ? "..." : ""}</p></div></div>`;
    }).join('');
    document.querySelectorAll('.project-card').forEach(card => {
        card.addEventListener('click', (e) => {
            const idx = card.dataset.idx;
            const proj = filtered[idx];
            openModal(proj);
        });
    });
}

function openModal(proj) {
    const modal = document.getElementById('projectModal');
    const data = proj[currentLang];
    modal.querySelector('.modal-image').style.backgroundImage = `url('${proj.img}')`;
    modal.querySelector('.modal-title').innerText = data.title;
    modal.querySelector('.modal-client').innerHTML = `<strong>${currentLang === 'fr' ? 'Client :' : 'Client:'}</strong> ${data.client}`;
    modal.querySelector('.modal-desc').innerHTML = `<strong>${currentLang === 'fr' ? 'Description :' : 'Description:'}</strong> ${data.longDesc}`;
    modal.querySelector('.modal-year').innerHTML = `<strong>${currentLang === 'fr' ? 'Année :' : 'Year:'}</strong> ${proj.year}`;
    modal.querySelector('.modal-location').innerHTML = `<strong>${currentLang === 'fr' ? 'Localisation :' : 'Location:'}</strong> ${data.location}`;
    modal.classList.add('active');
}

function initFilters() {
    const catBtns = document.querySelectorAll('#categoryFilters .filter-btn');
    const scopeContainer = document.getElementById('scopeFilters');
    const scopeBtns = document.querySelectorAll('#scopeFilters .sub-filter-btn');

    function updateActiveCategory(activeCat) {
        catBtns.forEach(btn => {
            if (btn.dataset.cat === activeCat) btn.classList.add('active');
            else btn.classList.remove('active');
        });
        currentCategory = activeCat;
        currentScope = null;
        scopeBtns.forEach(btn => btn.classList.remove('active'));
        renderProjects();
    }

    catBtns.forEach(btn => {
        btn.addEventListener('click', () => {
            updateActiveCategory(btn.dataset.cat);
        });
    });

    scopeBtns.forEach(btn => {
        btn.addEventListener('click', () => {
            const newScope = btn.dataset.scope;
            if (currentScope === newScope) {
                currentScope = null;
                btn.classList.remove('active');
            } else {
                currentScope = newScope;
                scopeBtns.forEach(b => b.classList.remove('active'));
                btn.classList.add('active');
            }
            renderProjects();
        });
    });

    scopeContainer.style.display = 'flex';
    updateActiveCategory('water');
}

// Langue
function setLanguage(lang) {
    currentLang = lang;
    document.querySelectorAll('[data-fr][data-en]').forEach(el => {
        el.innerText = lang === 'fr' ? el.getAttribute('data-fr') : el.getAttribute('data-en');
    });
    document.querySelectorAll('.lang-btn').forEach(btn => {
        if (btn.getAttribute('data-lang') === lang) btn.classList.add('active');
        else btn.classList.remove('active');
    });
    renderProjects();
    if (window.updateMapLanguage) window.updateMapLanguage(lang);
}

// Carte Afrique avec fond CartoDB Light (fiable)
let map;
function initMap() {
    map = L.map('africaMap').setView([5.0, 20.0], 4);
    
    // FOND DE CARTE STABLE : CartoDB Light (contient les frontières)
    L.tileLayer('https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png', {
        attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> &copy; CartoDB',
        subdomains: 'abcd',
        maxZoom: 19
    }).addTo(map);
    
    // Récupération des frontières africaines (fichier local recommandé, fallback GitHub)
    fetch('Afrique/africa.geojson')
        .then(res => res.ok ? res.json() : Promise.reject())
        .then(data => {
            L.geoJSON(data, {
                style: { color: "#4a6b8f", weight: 1.5, fillColor: "#eef3fc", fillOpacity: 0.3, opacity: 0.8 }
            }).addTo(map);
        })
        .catch(() => {
            console.warn("Fichier GeoJSON local non trouvé, utilisation d'un fallback externe.");
            fetch('https://raw.githubusercontent.com/datasets/geo-countries/master/data/countries.geojson')
                .then(res => res.json())
                .then(data => {
                    const africanNames = ["Algeria","Angola","Benin","Botswana","Burkina Faso","Burundi","Cabo Verde","Cameroon","Central African Republic","Chad","Comoros","Congo","Democratic Republic of the Congo","Djibouti","Egypt","Equatorial Guinea","Eritrea","Eswatini","Ethiopia","Gabon","Gambia","Ghana","Guinea","Guinea-Bissau","Ivory Coast","Kenya","Lesotho","Liberia","Libya","Madagascar","Malawi","Mali","Mauritania","Mauritius","Morocco","Mozambique","Namibia","Niger","Nigeria","Rwanda","Sao Tome and Principe","Senegal","Seychelles","Sierra Leone","Somalia","South Africa","South Sudan","Sudan","Tanzania","Togo","Tunisia","Uganda","Zambia","Zimbabwe"];
                    const africanGeoJSON = { type: "FeatureCollection", features: data.features.filter(f => africanNames.includes(f.properties?.ADMIN)) };
                    L.geoJSON(africanGeoJSON, { style: { color: "#4a6b8f", weight: 1.5, fillColor: "#eef3fc", fillOpacity: 0.3, opacity: 0.8 } }).addTo(map);
                })
                .catch(() => console.warn("Impossible de charger les frontières, mais le fond de carte reste visible."));
        });
    
    const starIcon = L.divIcon({
        html: '<div style="background: radial-gradient(circle, #ffd966, #ffb347); border-radius: 50%; width: 30px; height: 30px; display: flex; align-items: center; justify-content: center; box-shadow: 0 2px 8px rgba(0,0,0,0.4); border: 2px solid white;"><i class="fas fa-star" style="color: #FFD700; font-size: 18px; text-shadow: 0 0 3px #000;"></i></div>',
        className: 'custom-star-marker',
        iconSize: [30, 30],
        iconAnchor: [15, 15],
        popupAnchor: [0, -15]
    });
    const countriesData = computeInterventionCountries();
    countriesData.forEach(c => {
        const marker = L.marker([c.lat, c.lng], { icon: starIcon }).addTo(map);
        marker.bindPopup(`<b>${c.fr}</b><br>⭐ ${c.count} intervention(s) iGC`);
    });
    updateCountryLegend('fr');
    window.updateMapLanguage = function(lang) {
        updateCountryLegend(lang);
        map.eachLayer(layer => {
            if (layer instanceof L.Marker && layer.getPopup()) {
                const latlng = layer.getLatLng();
                const found = countriesData.find(c => Math.abs(c.lat - latlng.lat) < 0.5 && Math.abs(c.lng - latlng.lng) < 0.5);
                if (found) layer.getPopup().setContent(`<b>${lang === 'fr' ? found.fr : found.en}</b><br>⭐ ${found.count} intervention(s) iGC`);
            }
        });
    };
    
    // Forcer le recalcul de la taille de la carte après chargement complet
    setTimeout(() => map.invalidateSize(), 300);
    window.addEventListener('load', () => map.invalidateSize());
    window.addEventListener('resize', () => map.invalidateSize());
}
